          import math

class Par:

    def __init__( self, apertura, cierre ):
        self.apertura = apertura
        self.cierre = cierre

class SegmentTree:

    def __init__(self, string ):
        self.size = int ( 2 * math.pow(2, ( math.log2(len(string) )+ 1) ) )
        self.string = string
        self.st = [ Par( 0, 0 ) for i in range( self.size ) ]

        self.build( 0, 0, len(string)-1)

    def build(self, nodo, inicio, fin ):
        if inicio == fin:
            apertura = 0; cierre = 0
            if self.string[inicio] == "(":
                apertura = 1
            else:
                cierre = 1


            self.st[nodo].apertura = apertura
            self.st[nodo].cierre = cierre

        if inicio < fin:
            medio = ( inicio + fin ) // 2
            self.build( 2 * nodo + 1, inicio, medio )
            self.build( 2 * nodo + 2, medio + 1, fin )

            self.st[nodo].apertura = self.st[2 * nodo + 1].apertura\
                                     - min( self.st[2*nodo+1].apertura,self.st[2*nodo+2].cierre ) +\
                                     self.st[2 * nodo + 2].apertura
            self.st[nodo].cierre = self.st[2 * nodo + 2].cierre\
                                     - min( self.st[2*nodo+1].apertura,self.st[2*nodo+2].cierre ) +\
                                     self.st[2 * nodo + 1].cierre
