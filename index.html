              heapify(i){
                  if( ( ( i * 2 ) + 1 ) < this.size){
                      if( this.arr[2 * i + 1] < this.arr[i]){
                          this.swap(2 * i + 1, i);
                          this.heapify(2 * i + 1);
                      }

                  }
                  if( ( ( i * 2 ) + 2) < this.size){
                      if( this.arr[2 * i + 2] < this.arr[i]){
                          this.swap(2 * i + 2, i);
                          this.heapify(2 * i + 2);
                      }
                  }
              }

              heapifyInvertido(i){
                  if( i % 2 !== 0 && i !== 0){
                      if( this.arr[i] < parseInt( this.arr[( i -1 ) / 2] ) ){
                          this.swap(i, ( (i - 1 ) / 2) );
                          this.heapifyInvertido( ( i -1 )/ 2 );
                      }

                  }
                  if(  i % 2 === 0 && i !== 0){
                      if( this.arr[i] < parseInt( this.arr[( i - 2 ) / 2] ) ){
                          this.swap(i, ( i - 2) / 2 );
                          this.heapifyInvertido( ( i - 2) / 2 );
                      }
                  }
              }

                  this.arr[j] = aux;

                }
              }

            if(start < end){
                var middle = parseInt(( start + end ) / 2 );

                mergeSort(start, middle, arr);
                mergeSort( middle + 1, end, arr);

          }

          function merge(start, middle, end, arr) {
              var arrAux = [];
              var i = start;
              var j = middle + 1;

              while ( i <= middle && j <= end){
                  if( arr[i] < arr[j] ){
                      arrAux.push(arr[i]);
                      i += 1;
                  }else{
                      arrAux.push(arr[j]);
                       j += 1;
                  }
              }

              while( i <= middle){
                  arrAux.push(arr[i]);
                  i += 1;
              }

              while( j <= end){
                  arrAux.push(arr[j]);
                  j += 1;
              }

              var k = 0;
              while( start <= end ){
                  arr[start] = arrAux[k];
                  start += 1;
                  k += 1;
              }
              return arr;
